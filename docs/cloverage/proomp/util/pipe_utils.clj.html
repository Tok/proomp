001  (ns proomp.util.pipe-utils
002    (:require [proomp.config :as config]
003              [proomp.domain.prompt.prompt :as prom]
004              [proomp.domain.image.resolution :as res]
005              [proomp.domain.pipe.pipe-setup :as pipe-setup]
006              [cambium.core :as log]
007              [libpython-clj2.require :refer [require-python]]
008              [libpython-clj2.python :refer [py. py.-] :as py])
009    (:import (proomp.domain.image.resolution Resolution)
010             (proomp.domain.prompt.prompt Prompt)))
011  
012  (require-python 'torch '[torch.cuda :as cuda] 'transformers)
013  (require-python '[diffusers :refer [StableDiffusionPipeline StableDiffusionImg2ImgPipeline]])
014  
015  (defonce device "cuda")
016  
017  (defn- clear-cuda-cache []
018    (cuda/empty_cache)
019    (log/trace "Cuda cache cleared."))
020  
021  (defn- send-to-device [pipe] (py. pipe "to" device))
022  (defn- enable-attention-slicing [pipe]
023    (py. pipe "enable_attention_slicing")
024    (log/debug "Attention slicing enabled."))
025  
026  (defn- ->pipeline [type model-path]
027    (log/debug "Creating Pipeline.")
028    (clear-cuda-cache)
029    (let [pipe (py. type "from_pretrained" model-path
030                    :torch_dtype torch/float16)]
031      (send-to-device pipe)
032      (enable-attention-slicing pipe)
033      (log/trace {:pipe pipe})
034      pipe))
035  
036  (defn ->text-to-image-pipeline [] (->pipeline StableDiffusionPipeline config/model-path))
037  (defn ->image-to-image-pipeline [] (->pipeline StableDiffusionImg2ImgPipeline config/model-path))
038  
039  (defn- ->generator [seed] (py. (py/$c torch/Generator device) "manual_seed" seed))
040  
041  (defn- extract-first-image [result] (nth (py.- result :images) 0))
042  (defn generate-image [pipe ^Prompt prompt seed]
043    (let [^Resolution resolution res/active-image-resolution]
044      (extract-first-image
045        (py/$c pipe (prom/full-prompt prompt)
046               :negative_prompt (prom/full-negative-prompt prompt)
047               :generator (->generator seed)
048               :height (:h resolution) :width (:w resolution)
049               :num_inference_steps (:iterations pipe-setup/image-pipe-setup)
050               :guidance_scale (:scale pipe-setup/image-pipe-setup)))))
051  
052  (defn generate-i2i [pipe ^Prompt prompt seed init-image]
053    (extract-first-image
054      (py/$c pipe (prom/full-prompt prompt)
055             :negative_prompt (prom/full-negative-prompt prompt)
056             :init_image init-image
057             :strength (:noise pipe-setup/i2i-pipe-setup)
058             :generator (->generator seed)
059             :num_inference_steps (:iterations pipe-setup/i2i-pipe-setup)
060             :guidance_scale (:scale pipe-setup/i2i-pipe-setup))))
