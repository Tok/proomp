028  (defn save-py-image!
029    ([image file-name] (save-py-image! image file-name false))
030    ([image file-name overwrite?]
031     (if overwrite?
032       (do-py-image-save! image file-name)
033       (if (file-utils/file-exists? file-name)
034         (log/warn {"File exists. Skipping." file-name})
035         (do-py-image-save! image file-name)))))

077  (defn ^RenderedImage scale [image factor]
078    (let [w (.getWidth image)
079          h (.getHeight image)
080          type (.getType image)
081          scaled-w (* factor w)
082          scaled-h (* factor h)
083          scaled (BufferedImage. scaled-w scaled-h type)
084          ctx (.createGraphics scaled)
085          render-key RenderingHints/KEY_INTERPOLATION
086          render-value RenderingHints/VALUE_INTERPOLATION_BILINEAR]
087      (.setRenderingHint ctx render-key render-value)
088      (.drawImage ctx image 0 0 scaled-w scaled-h 0 0 w h nil)
089      (.dispose ctx)
090      scaled))

098    (let [temp-dir (new File config/temp-media-path)]
099      (.mkdir temp-dir)
100      (let [file (File/createTempFile "temp" ".png" temp-dir)
101            file-name (.getPath file)]                        ;todo rewrite without temp file.
102        (save-py-image! pil-image file-name true)
103        (let [buffered-image (open-buffered-image file-name)]
104          (try
105            (io/delete-file file-name)
106            (catch IOException e (.getMessage e)))
107          buffered-image))))

111  (defn zoom-center [pil-image zoom]
112    (let [w (pil->w pil-image)
113          h (pil->h pil-image)
114          x (* w 0.5)
115          y (* h 0.5)
116          cropped-width (/ w zoom)
117          cropped-height (/ h zoom)
118          left (int (- x (/ cropped-width 2)))
119          top (int (- y (/ cropped-height 2)))
120          right (int (+ x (/ cropped-width 2)))
121          bottom (int (+ y (/ cropped-height 2)))
122          cropped-image (crop pil-image left top right bottom)]
123      (resize cropped-image w h)))
