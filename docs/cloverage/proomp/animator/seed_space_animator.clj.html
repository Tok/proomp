001  (ns proomp.animator.seed-space-animator
002    (:require
003      [cambium.core :as log]
004      [proomp.config :as config]
005      [clojure.java.io :as io]
006      [libpython-clj2.python :refer [py.] :as py]
007      [libpython-clj2.require :refer [require-python]]
008      [proomp.domain.image.resolution :as res]
009      [proomp.domain.pipe.pipe-setup :as pipe-setup]
010      [proomp.util.file-utils :as file-utils]
011      [proomp.util.image-utils :as image-utils]
012      [proomp.util.pipe-utils :as pipe-utils]))
013  
014  (require-python 'torch '[torch.cuda :as cuda] 'transformers)
015  (require-python '[PIL.Image :refer [open new]])
016  (require-python '[PIL.ImageChops :as chops])
017  
018  (def rotation-degree -0.0)                                  ;per frame
019  (def x-offset -0)                                           ;pixels per frame
020  (def y-offset 0)                                            ;pixels per frame
021  (def zoom 1.000)                                            ;should be >= 1.000
022  (def apply-transformations? false)
023  (def apply-color-correction? false)
024  
025  (defn- initial-frame [prompt start-seed initial-frame-file-name]
026    (let [resolution res/active-animation-resolution]
027      (if (not (file-utils/file-exists? initial-frame-file-name))
028        (log/error {:initial-frame-missing initial-frame-file-name})
029        (py/with [image (image-utils/open-py-image initial-frame-file-name)]
030                 (let [resized (image-utils/resize image (:w resolution) (:h resolution))]
031                   (image-utils/save-py-image! resized initial-frame-file-name)
032                   (log/info {:initial-frame-saved initial-frame-file-name})
033                   resized)))))
034  
035  (defn- apply-transformations [pil-image]
036    (let [rotated (py. pil-image "rotate" rotation-degree)
037          chopped (chops/offset rotated x-offset y-offset)]
038      (image-utils/zoom-center chopped zoom)))
039  
040  (defn- generate-image! [pil-image reference-image pipe prompt seed frame-file-name first-image?]
041    (let [pic (if (and (not first-image?) apply-transformations?)
042                (apply-transformations pil-image) pil-image)]
043      (let [corrected (if apply-color-correction? (image-utils/fix-colors pic reference-image) pic)
044            sample (pipe-utils/generate-i2i pipe prompt seed corrected)]
045        (image-utils/save-py-image! sample frame-file-name)
046        sample)))
047  
048  (def last-frame (atom (image-utils/->pil-image 1 1)))       ;todo refactor to remove atom
049  (defn- generate-frames [pil-image ref-image pipe prompt start-seed frame-count]
050    (reset! last-frame pil-image)
051    (doseq [frame-number (range 0 frame-count)]
052      (let [new-seed (+ start-seed frame-number)
053            iterations (:iterations pipe-setup/i2i-pipe-setup)
054            scale (:scale pipe-setup/i2i-pipe-setup)
055            frame-file-name (file-utils/frame-name (:text prompt) new-seed iterations scale)
056            first-image? (= frame-number 0)]
057        (if (file-utils/file-exists? frame-file-name)
058          (do
059            (log/warn {:skip-existing frame-file-name})
060            (reset! last-frame (image-utils/open-py-image frame-file-name)))
061          (let [next-frame (generate-image! @last-frame ref-image pipe prompt new-seed frame-file-name first-image?)]
062            (reset! last-frame next-frame))))))
063  
064  (defn animate [pipe prompt start-seed frame-count]
065    (log/info {:resolution res/active-animation-resolution :frame-count frame-count})
066    (let [result-dir (file-utils/animation-frame-dir (:text prompt))]
067      (log/debug {:result-path result-dir})
068      (io/make-parents result-dir)
069      (let [initial-frame-file-name (file-utils/file-name (:text prompt) start-seed)
070            first-image (initial-frame prompt start-seed initial-frame-file-name)]
071        (log/info "Loading reference image.")
072        (let [ref-image (image-utils/prepare-reference-image first-image)]
073          (generate-frames first-image ref-image pipe prompt start-seed frame-count)))))
