001  (ns proomp.domain.animation.frame-transformation
002    (:import (clojure.lang MapEntry)))
003  
004  ;;Transformations are applied between generating animation frames.
005  (defrecord FrameTransformation
006    [zoom rotation-degree x-offset-pixels y-offset-pixels apply-color-correction?])
007  
008  (def ^:private transformation-templates
009    {::none                       (->FrameTransformation 1.0 0.0 0 0 false)
010     ::color-corrected            (map->FrameTransformation {:apply-color-correction? true})
011     ::slow-zoom                  (map->FrameTransformation {:zoom 1.020})
012     ::default-zoom               (map->FrameTransformation {:zoom 1.050})
013     ::fast-zoom                  (map->FrameTransformation {:zoom 1.100})
014     ::clockwise-rotation         (map->FrameTransformation {:rotation-degree -1.0})
015     ::counter-clockwise-rotation (map->FrameTransformation {:rotation-degree 1.0})
016     ::move-left                  (map->FrameTransformation {:x-offset-pixels -5})
017     ::move-right                 (map->FrameTransformation {:x-offset-pixels 5})
018     ::move-up                    (map->FrameTransformation {:y-offset-pixels -5})
019     ::move-down                  (map->FrameTransformation {:y-offset-pixels 5})})
020  
021  (defn- merge-with-default [entry]
022    (let [[k v] entry
023          zoom (min (or (:zoom v) 1.000) 1.000)
024          rd (:rotation-degree v)
025          deg (if (not (nil? rd)) rd 0.0)
026          rot (mod deg 360.0)
027          x-off (or (:x-offset-pixels v) 0)
028          y-off (or (:y-offset-pixels v) 0)
029          correct? (if (:apply-color-correction? v) true false)]
030      (MapEntry. k (->FrameTransformation zoom rot x-off y-off correct?))))
031  
032  (defonce transformations (map merge-with-default transformation-templates))
033  (def active-transformation (::slow-zoom transformations))
034  
035  (def apply-transformations? (not (= active-transformation (::none transformations))))
