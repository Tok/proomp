001  (ns proomp.domain.animation.transformation
002    (:import (clojure.lang MapEntry)))
003  
004  (defrecord Transformation [zoom rotation-degree x-offset-pixels y-offset-pixels])
005  (def ^:private transformation-templates
006    {:none                       (->Transformation 1.0 0.0 0 0)
007     :slow-zoom                  (map->Transformation {:zoom 1.020})
008     :default-zoom               (map->Transformation {:zoom 1.050})
009     :fast-zoom                  (map->Transformation {:zoom 1.100})
010     :clockwise-rotation         (map->Transformation {:rotation-degree -1.0})
011     :counter-clockwise-rotation (map->Transformation {:rotation-degree 1.0})
012     :move-left                  (map->Transformation {:x-offset-pixels -5})
013     :move-right                 (map->Transformation {:x-offset-pixels 5})
014     :move-up                    (map->Transformation {:y-offset-pixels -5})
015     :move-down                  (map->Transformation {:y-offset-pixels 5})})
016  
017  (defn- merge-with-default [entry]
018    (let [[k v] entry
019          zoom (min (or (:zoom v) 1.000) 1.000)
020          rot (mod (or (:rotation-degree v) 0.0) 360.0)
021          x-off (or (:x-offset-pixels v) 0)
022          y-off (or (:y-offset-pixels v) 0)]
023      (MapEntry. k (->Transformation zoom rot x-off y-off))))
024  
025  (defonce transformations (map merge-with-default transformation-templates))
