001  (ns proomp.core
002    (:require [proomp.config :as config]                      ;don't remove this
003              [proomp.animator.seed-space-animator :as seed-space-animator]
004              [proomp.domain.prompt.prompt :as prompt]
005              [proomp.util.file-utils :as file-utils]
006              [proomp.util.image-utils :as image-utils]
007              [proomp.util.pipe-utils :as pipe-utils]
008              [proomp.util.video-utils :as video-utils]
009              [cambium.core :as log]))
010  
011  (defonce ^:private modes [::animation ::images ::video])
012  (defonce active-mode ::images)
013  ;(defonce active-mode ::animation)
014  ;(defonce active-mode ::video)
015  
016  (defonce text "Caliper Remote")
017  (defonce negative-text "")
018  (defonce additions "((photorealistic)) (photo) sharp focused")
019  (defonce negative-additions "((blurry)) (drawing) grayscale deformed disfigured")
020  (defonce full-prompt (prompt/->Prompt text negative-text additions negative-additions))
021  
022  (defonce start-seed 0)
023  (defonce number-of-images-to-generate 1000)
024  
025  (defonce animation-start-seed 0)                            ;choose a good seed by generating ::images first
026  
027  (defn- do-generation! [pipe seed file-name]
028    (let [image (pipe-utils/generate-image pipe full-prompt seed)]
029      (image-utils/save-py-image! image file-name)
030      image))
031  
032  (defn- generate-image! [pipe seed]
033    (log/trace {:seed seed})
034    (let [file-name (file-utils/file-name (:text full-prompt) seed)]
035      (if (not (file-utils/file-exists? file-name))
036        (do-generation! pipe seed file-name)
037        (log/warn {:skip-existing file-name}))))
038  
039  (defn -main []
040    (if (= active-mode ::animation)
041      (let [pipe (pipe-utils/->image-to-image-pipeline)
042            frame-count (* 3600 5)]
043        (seed-space-animator/animate pipe full-prompt animation-start-seed frame-count))
044      (if (= active-mode ::images)
045        (let [pipe (pipe-utils/->text-to-image-pipeline)]
046          (doseq [seed (range start-seed (+ start-seed number-of-images-to-generate))]
047            (generate-image! pipe seed)))
048        (video-utils/generate-video-from-frames (:text full-prompt)))))
